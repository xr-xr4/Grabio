ğŸ› ï¸ Usage (Python)

from grabio import Grabio

# Initialize
g = Grabio('https://example.com')

# Extract specific info
print("IP Address:", g.ip())
print("WHOIS Info:", g.whois_info())
print("SSL Info:", g.ssl_info())

# Extract full info
full_info = g.info()
print(full_info)

# Save info as JSON
g.show_info()
